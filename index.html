function diffYMDHMS(from, to) { let start = new Date(from.getTime()); let end = new Date(to.getTime()); let years = end.getFullYear() - start.getFullYear(); let months = end.getMonth() - start.getMonth(); let days = end.getDate() - start.getDate(); let hours = end.getHours() - start.getHours(); let minutes = end.getMinutes() - start.getMinutes(); let seconds = end.getSeconds() - start.getSeconds(); // معالجة الثواني if (seconds < 0) { seconds += 60; minutes--; } // معالجة الدقايق if (minutes < 0) { minutes += 60; hours--; } // معالجة الساعات if (hours < 0) { hours += 24; days--; } // معالجة الأيام (دي أهم حتة عشان السوالب) if (days < 0) { // بنجيب آخر يوم في الشهر اللي فات let lastMonth = new Date(end.getFullYear(), end.getMonth(), 0); days += lastMonth.getDate(); months--; } // معالجة الشهور if (months < 0) { months += 12; years--; } return { years, months, days, hours, minutes, seconds }; }
